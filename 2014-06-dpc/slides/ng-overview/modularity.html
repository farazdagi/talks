Modularity

Define your service:

// app/scripts/services/test.js
angular.module('demoApp')
  .service('Test', function Test() {
    this.yell = function () {
      console.log('hello!');
    };
  });

Use the service:

// app/scripts/controllers/main.js
angular.module('demoApp')
  .controller('MainCtrl', function ($scope, Test) {
    Test.yell();
  });
