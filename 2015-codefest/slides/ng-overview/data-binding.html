Data Binding

Setup view:

<div class="jumbotron">
    <p ng-show="name">Hello, {{ name }}</p>
    <input type="text" ng-model="name"/>
</div>

Setup scope:

// app/scripts/controllers/main.js
angular.module('demoApp')
  .controller('MainCtrl', function ($scope) {
    $scope.name = 'Alex';
  });
